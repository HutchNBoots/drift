# Drift POC Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a playable browser-based POC of Drift — a one-tap gravity flip lofi rhythm platformer with synthesised audio, beat-synced obstacles, and a feel system that makes skilled play physically distinct.

**Architecture:** All game state lives in a single plain JS object. Each module exports pure functions (`module.update(state) → state`) that main.js orchestrates in sequence each frame. The renderer reads state and draws — no logic, no mutation. Tests run in Node via Vitest against the pure logic modules only.

**Tech Stack:** Vanilla JS (ES modules), HTML5 Canvas, Web Audio API, Web Vibration API, Vitest (tests only)

```js
// src/level.js
import { precomputeObstacles } from './obstacles.js';

export function loadLevel(json) {
  const obstacles = precomputeObstacles(json);
  return {
    level: json,
    obstacles,
    currentSection: json.sections[0],
    beat: {
      current: 0,
      startTime: 0,
      bpm: json.bpm,
      interval: 60 / json.bpm,
    },
  };
}

export function resolveSection(state) {
  const { beat, level, currentSection } = state;
  if (beat.current < currentSection.endBeat) return state;

  const currentIndex = level.sections.findIndex(
    s => s.startBeat === currentSection.startBeat
  );
  const nextSection = level.sections[currentIndex + 1];
  if (!nextSection) return state;

  return { ...state, currentSection: nextSection, status: 'section_clear' };
}
```

```js
// src/obstacles.js
import {
  CANVAS_HEIGHT,
  OBSTACLE_HEIGHT_DEFAULT,
  OBSTACLE_WIDTH_DEFAULT,
} from './constants.js';

export function precomputeObstacles(level) {
  return level.obstacles.map(obs => {
    const section = level.sections.find(
      s => obs.beat >= s.startBeat && obs.beat < s.endBeat
    );
    const sectionOffset = level.sections
      .filter(s => s.startBeat < section.startBeat)
      .reduce((sum, s) => sum + (s.endBeat - s.startBeat) * s.speed, 0);

    const worldX = sectionOffset + (obs.beat - section.startBeat) * section.speed;
    const width = obs.width ?? OBSTACLE_WIDTH_DEFAULT;
    const height = obs.height ?? OBSTACLE_HEIGHT_DEFAULT;

    return {
      x: worldX,
      y: obs.origin === 'floor' ? CANVAS_HEIGHT : 0,
      width,
      height,
      origin: obs.origin,
      active: false,
      beaten: false,
      proximityBand: null,
    };
  });
}

export function updateObstacles(state) {
  return { state, proximityEvents: [] };
}
```

```js
import {
  CANVAS_HEIGHT,
  CANVAS_WIDTH,
  OBSTACLE_HEIGHT_DEFAULT,
  OBSTACLE_WIDTH_DEFAULT,
  PLAYER_X,
  PLAYER_RADIUS,
  PROXIMITY_PIXEL,
  PROXIMITY_CLOSE,
  PROXIMITY_NEAR,
} from './constants.js';

// ... (keep precomputeObstacles as-is)

export function updateObstacles(state) {
  const { obstacles, scrollX, player } = state;
  const proximityEvents = [];

  const updated = obstacles.map(obs => {
    const screenX = obs.x - scrollX;

    // Activate when entering the viewport
    if (!obs.active && screenX < CANVAS_WIDTH + obs.width) {
      obs = { ...obs, active: true };
    }

    // Mark beaten when obstacle's right edge passes player's left edge
    if (obs.active && !obs.beaten && screenX + obs.width < PLAYER_X - PLAYER_RADIUS) {
      const band = calcProximityBand(obs, player.y);
      proximityEvents.push(band);
      obs = { ...obs, beaten: true, proximityBand: band };
    }

    return obs;
  });

  return { state: { ...state, obstacles: updated }, proximityEvents };
}

function calcProximityBand(obs, playerY) {
  let gap;
  if (obs.origin === 'floor') {
    const obstacleTop = obs.y - obs.height;    // CANVAS_HEIGHT - height
    gap = obstacleTop - (playerY + PLAYER_RADIUS);
  } else {
    const obstacleBottom = obs.y + obs.height; // 0 + height
    gap = (playerY - PLAYER_RADIUS) - obstacleBottom;
  }
  gap = Math.max(0, gap);

  if (gap < PROXIMITY_PIXEL) return 'pixel';
  if (gap < PROXIMITY_CLOSE) return 'close';
  if (gap <= PROXIMITY_NEAR) return 'near';
  return 'safe';
}
```

```js
// src/player.js
import {
  GRAVITY,
  MAX_FALL_SPEED,
  CANVAS_HEIGHT,
  PLAYER_RADIUS,
  PLAYER_X,
  TRAIL_MAX_LENGTH,
} from './constants.js';

export function updatePlayer(state, dt, now = performance.now()) {
  const { player, status } = state;
  if (status !== 'playing') return state;

  // Apply gravity
  let vy = player.vy + GRAVITY * player.gravityDir;
  if (vy > MAX_FALL_SPEED) vy = MAX_FALL_SPEED;
  if (vy < -MAX_FALL_SPEED) vy = -MAX_FALL_SPEED;

  // Decrement coyote and input buffer counters
  let { inputBuffer, coyoteFrames, gravityDir } = player;
  if (coyoteFrames > 0) coyoteFrames--;
  if (inputBuffer > 0) {
    inputBuffer--;
    if (inputBuffer === 0) {
      vy = 0;
      gravityDir = gravityDir * -1;
    }
  }

  const y = player.y + vy;

  // Update trail: prepend current position, age all, remove expired
  const trail = [
    { x: PLAYER_X, y: player.y, age: 0 },
    ...player.trail
      .map(t => ({ ...t, age: t.age + 1 }))
      .filter(t => t.age < TRAIL_MAX_LENGTH),
  ];

  const newPlayer = { ...player, y, vy, gravityDir, inputBuffer, coyoteFrames, trail };
  const newState = { ...state, player: newPlayer };

  // Death: off-screen
  if (y - PLAYER_RADIUS < 0 || y + PLAYER_RADIUS > CANVAS_HEIGHT) {
    return { ...newState, status: 'dead', deathTime: now };
  }

  // Death: obstacle collision
  for (const obs of state.obstacles) {
    if (obs.active && overlapsObstacle(obs, state.scrollX, y)) {
      return { ...newState, status: 'dead', deathTime: now };
    }
  }

  return newState;
}

export function handleTap(state) {
  const { player, status } = state;
  if (status !== 'playing') return state;

  // Coyote time: flip immediately
  if (player.coyoteFrames > 0) {
    return {
      ...state,
      player: {
        ...player,
        vy: 0,
        gravityDir: player.gravityDir * -1,
        coyoteFrames: 0,
        inputBuffer: 0,
      },
    };
  }

  // Normal flip
  return {
    ...state,
    player: {
      ...player,
      vy: 0,
      gravityDir: player.gravityDir * -1,
      inputBuffer: 0,
    },
  };
}

function overlapsObstacle(obs, scrollX, playerY) {
  const screenX = obs.x - scrollX;
  const playerLeft = PLAYER_X - PLAYER_RADIUS;
  const playerRight = PLAYER_X + PLAYER_RADIUS;
  const playerTop = playerY - PLAYER_RADIUS;
  const playerBottom = playerY + PLAYER_RADIUS;

  const obsLeft = screenX;
  const obsRight = screenX + obs.width;
  const obsTop = obs.origin === 'floor' ? CANVAS_HEIGHT - obs.height : 0;
  const obsBottom = obs.origin === 'floor' ? CANVAS_HEIGHT : obs.height;

  return playerRight > obsLeft && playerLeft < obsRight &&
         playerBottom > obsTop && playerTop < obsBottom;
}
```

```js
// src/flow.js
import {
  FLOW_GAIN_PIXEL,
  FLOW_GAIN_CLOSE,
  FLOW_DRAIN_SAFE,
  FLOW_GAIN_SECTION_CLEAR,
  FLOW_STATE_MID,
  FLOW_STATE_HIGH,
} from './constants.js';

export function updateFlow(state, proximityEvents) {
  let value = state.flow.value;

  if (state.status === 'dead') {
    value = 0;
  } else if (state.status === 'section_clear') {
    value = Math.min(100, value + FLOW_GAIN_SECTION_CLEAR);
  } else {
    for (const event of proximityEvents) {
      if (event === 'pixel') value += FLOW_GAIN_PIXEL;
      else if (event === 'close') value += FLOW_GAIN_CLOSE;
      else if (event === 'safe') value -= FLOW_DRAIN_SAFE;
      // 'near' → no change
    }
  }

  value = Math.max(0, Math.min(100, value));
  const flowState = value >= FLOW_STATE_HIGH ? 'high' : value >= FLOW_STATE_MID ? 'mid' : 'low';

  return { ...state, flow: { value, state: flowState } };
}
```

```js
// --- Synthesised lofi bed ---

function startLofibed(bpm) {
  const beatInterval = 60 / bpm;
  let beat = 0;

  function tick() {
    const t = audioCtx.currentTime;
    scheduleKick(t, beat, beatInterval);
    scheduleSnare(t, beat, beatInterval);
    scheduleHihat(t, beat, beatInterval);
    scheduleBass(t, beat, beatInterval);
    scheduleChord(t, beat, beatInterval);
    beat++;
    // Schedule next tick slightly early for reliability
    setTimeout(tick, (beatInterval * 1000) - 20);
  }
  tick();
}

function scheduleKick(t, beat, interval) {
  if (beat % 2 !== 0) return; // beats 0, 2 (= beats 1, 3 in 1-indexed)
  const osc = audioCtx.createOscillator();
  const gain = audioCtx.createGain();
  osc.connect(gain);
  gain.connect(masterGain);
  osc.frequency.setValueAtTime(160, t);
  osc.frequency.exponentialRampToValueAtTime(60, t + 0.2);
  gain.gain.setValueAtTime(1, t);
  gain.gain.exponentialRampToValueAtTime(0.001, t + 0.2);
  osc.start(t);
  osc.stop(t + 0.2);
}

function scheduleSnare(t, beat, interval) {
  if (beat % 2 !== 1) return; // beats 1, 3 (= beats 2, 4 in 1-indexed)
  const bufferSize = audioCtx.sampleRate * 0.04;
  const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
  const data = buffer.getChannelData(0);
  for (let i = 0; i < bufferSize; i++) data[i] = (Math.random() * 2 - 1);

  const source = audioCtx.createBufferSource();
  source.buffer = buffer;

  const filter = audioCtx.createBiquadFilter();
  filter.type = 'bandpass';
  filter.frequency.value = 2000;
  filter.Q.value = 0.5;

  const gain = audioCtx.createGain();
  gain.gain.setValueAtTime(0.6, t);
  gain.gain.exponentialRampToValueAtTime(0.001, t + 0.04);

  source.connect(filter);
  filter.connect(gain);
  gain.connect(masterGain);
  source.start(t);
  source.stop(t + 0.04);
}

function scheduleHihat(t, beat, interval) {
  // Two hi-hats per beat: on the beat and the half-beat
  [0, 0.5].forEach(offset => {

    filter.frequency.value = 8000;
```

```js
    ctx.arc(cx, cy, r, 0, Math.PI * 2);
    ctx.stroke();
  }
}

function drawObstacles(ctx, obstacles, scrollX) {
  ctx.fillStyle = COLOR_OBSTACLE;
  for (const obs of obstacles) {
    if (!obs.active) continue;
    const screenX = obs.x - scrollX;
    if (screenX + obs.width < 0 || screenX > CANVAS_WIDTH) continue;

    const y = obs.origin === 'floor' ? CANVAS_HEIGHT - obs.height : 0;
    ctx.fillRect(screenX, y, obs.width, obs.height);
  }
}

function drawTrail(ctx, player, flow) {
  if (player.trail.length < 2) return;

  const isHigh = flow.state === 'high';
  const trailWidth = isHigh ? 8 : 4;

  for (let i = 0; i < player.trail.length - 1; i++) {
    const t = player.trail[i];
    const alpha = 1 - t.age / TRAIL_MAX_LENGTH;
    ctx.strokeStyle = isHigh
      ? `rgba(240, 100, 20, ${alpha * 0.8})`
      : `rgba(240, 153, 123, ${alpha * 0.6})`;
    ctx.lineWidth = trailWidth * (1 - t.age / TRAIL_MAX_LENGTH);
    ctx.beginPath();
    ctx.moveTo(player.trail[i].x, player.trail[i].y);
    ctx.lineTo(player.trail[i + 1].x, player.trail[i + 1].y);
    ctx.stroke();
  }
}

function drawPlayer(ctx, player, flow) {
  ctx.fillStyle = COLOR_PLAYER;
  ctx.beginPath();
  ctx.arc(PLAYER_X, player.y, PLAYER_RADIUS, 0, Math.PI * 2);
  ctx.fill();

  // Vinyl label motif — small inner circle
  ctx.fillStyle = 'rgba(245, 239, 224, 0.5)';
  ctx.beginPath();
  ctx.arc(PLAYER_X, player.y, PLAYER_RADIUS * 0.35, 0, Math.PI * 2);
  ctx.fill();
}

function drawFlowVignette(ctx, flow) {
  if (flow.state === 'low') return;

  const intensity = flow.state === 'high' ? 0.35 : 0.18;
  const gradient = ctx.createRadialGradient(
    CANVAS_WIDTH / 2, CANVAS_HEIGHT / 2, CANVAS_HEIGHT * 0.3,
    CANVAS_WIDTH / 2, CANVAS_HEIGHT / 2, CANVAS_WIDTH * 0.75
  );
  gradient.addColorStop(0, 'rgba(240, 153, 123, 0)');
  gradient.addColorStop(1, `rgba(240, 153, 123, ${intensity})`);

  ctx.fillStyle = gradient;
  ctx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);
}

function drawHUD(ctx, flow, state) {
  // Flow meter bar — top-left
  const barW = 120;
  const barH = 8;
  const bx = 16;
  const by = 16;

  ctx.fillStyle = 'rgba(42, 31, 10, 0.2)';
  ctx.fillRect(bx, by, barW, barH);

  const fillColour = flow.state === 'high' ? '#F0997B'
                   : flow.state === 'mid'  ? '#D85A30'
                   :                         '#993C1D';
  ctx.fillStyle = fillColour;
  ctx.fillRect(bx, by, barW * (flow.value / 100), barH);
}

function drawOverlays(ctx, status, deathTime, now, state) {
  if (status === 'dead') {
    const elapsed = deathTime ? now - deathTime : 0;
    const alpha = Math.min(0.6, elapsed / 400);
    ctx.fillStyle = `rgba(245, 239, 224, ${alpha})`;
    ctx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);

    if (elapsed >= DEATH_RETRY_MS) {
      ctx.fillStyle = COLOR_UI_TEXT;
      ctx.font = '20px monospace';
      ctx.textAlign = 'center';
      ctx.fillText('tap to retry', CANVAS_WIDTH / 2, CANVAS_HEIGHT / 2 + 8);
    }
  }

  if (status === 'section_clear') {
    const sectionClearStart = state.sectionClearTime || now;
    const elapsed = now - sectionClearStart;
    const alpha = Math.max(0, 1 - elapsed / SECTION_CLEAR_DISPLAY_MS);

    ctx.fillStyle = `rgba(216, 90, 48, ${alpha * 0.15})`;
    ctx.fillRect(0, 0, CANVAS_WIDTH, CANVAS_HEIGHT);

    ctx.fillStyle = `rgba(42, 31, 10, ${alpha})`;
    ctx.font = '18px monospace';
    ctx.textAlign = 'center';
    ctx.fillText(
      state.currentSection?.label ? `${state.currentSection.label} cleared` : 'Section cleared',
      CANVAS_WIDTH / 2,
      CANVAS_HEIGHT / 2
    );
  }
}
```

**Files:**
- Modify: `src/main.js`

- [ ] **Step 1: Write main.js**

```js
// src/main.js
import { loadLevel, resolveSection } from './level.js';
import { updateObstacles } from './obstacles.js';
import { updatePlayer, handleTap } from './player.js';
import { updateFlow } from './flow.js';
import { initAudio, updateBeatClock, playSFX } from './audio.js';
import { vibrate } from './haptics.js';
import { draw } from './renderer.js';
import { CANVAS_WIDTH, CANVAS_HEIGHT, PLAYER_X, PLAYER_RADIUS, DEATH_RETRY_MS } from './constants.js';

const canvas = document.getElementById('game');
canvas.width = CANVAS_WIDTH;
canvas.height = CANVAS_HEIGHT;
const ctx = canvas.getContext('2d');

let state = null;
let lastTimestamp = 0;

async function boot() {
  const res = await fetch('../levels/world1.json');
  const levelJson = await res.json();
  state = buildInitialState(levelJson);
  requestAnimationFrame(gameLoop);
}

function buildInitialState(levelJson) {
  const { level, obstacles, currentSection, beat } = loadLevel(levelJson);
  return {
    player: {
      x: PLAYER_X,
      y: CANVAS_HEIGHT / 2,
      vy: 0,
      gravityDir: 1,
      trail: [],
      inputBuffer: 0,
      coyoteFrames: 0,
    },
    scrollX: 0,
    obstacles,
    beat,
    flow: { value: 0, state: 'low' },
    level,
    currentSection,
    status: 'playing',
    deathTime: null,
    sectionClearTime: null,
    inputEnabled: true,
  };
}

function gameLoop(timestamp) {
  const dt = timestamp - lastTimestamp;
  lastTimestamp = timestamp;

  if (state.status === 'playing') {
    state = updateBeatClock(state);
    state = resolveSection(state);
    state = updateScroll(state);
    state = updatePlayer(state, dt);
    const { state: afterObstacles, proximityEvents } = updateObstacles(state);
    state = afterObstacles;
    state = updateFlow(state, proximityEvents);
    handleProximityFeedback(proximityEvents);
  }

  if (state.status === 'section_clear') {
    if (!state.sectionClearTime) {
      state = { ...state, sectionClearTime: performance.now() };
      playSFX('section_clear');
      vibrate('section_clear');
    }
    const elapsed = performance.now() - state.sectionClearTime;
    if (elapsed >= 600) {
      state = { ...state, status: 'playing', sectionClearTime: null };
    }
  }

  draw(ctx, state);
  requestAnimationFrame(gameLoop);
}

function updateScroll(state) {
  const pixelsPerFrame = state.currentSection.speed * (state.beat.bpm / 60) / 60;
  return { ...state, scrollX: state.scrollX + pixelsPerFrame };
}

function handleProximityFeedback(proximityEvents) {
  for (const event of proximityEvents) {
    if (event === 'pixel') { playSFX('pixel_pass'); vibrate('pixel_pass'); }
    else if (event === 'close') { playSFX('close_pass'); }
  }
}

function onTap() {
  if (!state) return;

  if (state.status === 'dead') {
    const elapsed = performance.now() - state.deathTime;
    if (elapsed >= DEATH_RETRY_MS) {
      const levelJson = state.level;
      const beatState = initAudio(levelJson.bpm);
      const { level, obstacles, currentSection } = loadLevel(levelJson);
      state = {
        player: { x: PLAYER_X, y: CANVAS_HEIGHT / 2, vy: 0, gravityDir: 1, trail: [], inputBuffer: 0, coyoteFrames: 0 },
        scrollX: 0,
        obstacles,
        beat: beatState,
        flow: { value: 0, state: 'low' },
        level,
        currentSection,
        status: 'playing',
        deathTime: null,
        sectionClearTime: null,
        inputEnabled: true,
      };
    }
    return;
  }

  if (state.status === 'playing') {
    state = handleTap(state);
    playSFX('flip');
  }
}

// Start audio on first tap (browser autoplay policy)
let audioStarted = false;
canvas.addEventListener('pointerdown', () => {
  if (!audioStarted) {
    audioStarted = true;
    const beatState = initAudio(state.level.bpm);
    state = { ...state, beat: beatState };
  }
  onTap();
});

boot();
```
